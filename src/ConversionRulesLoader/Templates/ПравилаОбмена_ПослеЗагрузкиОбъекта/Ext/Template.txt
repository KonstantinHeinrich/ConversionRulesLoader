	// ПараметрыОбъекта = Неопределено.
	ОбъектМодифицирован = Ложь; // Булево. Флажок указывает на то, что в обработчике объект был изменен. По умолчанию
	// содержит значение Истина. То есть обработка загрузки считает, что если выполнялся какой-либо обработчик, то

	НаборЗаписей = Неопределено; // РегистрСведенийНаборЗаписей. Загруженный набор записей. Только при загрузке в
